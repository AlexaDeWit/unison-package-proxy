npmDistEncoder : NpmDist -> Json
npmDistEncoder = cases
  (NpmDist tarball shasum integrity fileCount unpackedSize signatures) ->
    use object empty
    empty
      |> addText "tarball" tarball
      |> addText "shasum" shasum
      |> addOptionalText "integrity" integrity
      |> addOptionalNat "fileCount" fileCount
      |> addOptionalNat "unpackedSize" unpackedSize
      |> addOptionalList "signatures" (Optional.map (map npmSignatureEncoder) signatures)

npmAcceptFormatDecoder : '{Decoder} NpmAcceptFormat
npmAcceptFormatDecoder =
  Decoder.text |> Decoder.andThen (cases

    "full" -> Decoder.succeed NpmAcceptFormat.Full
