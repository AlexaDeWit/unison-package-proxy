namespace NpmRegistryTypes.Json

{{ Encode an NpmHuman to JSON. }}
npmHumanEncoder : NpmHuman -> Json
npmHumanEncoder = cases
  (NpmHuman name email url) ->
    use Json.object addOptionalText
    empty
      |> addOptionalText "name" name
      |> addOptionalText "email" email
      |> addOptionalText "url" url

{{ Decode an NpmHuman from JSON. }}
npmHumanDecoder : '{Decoder} NpmHuman
npmHumanDecoder = do
  use object at!
  use object optionalAt!
  name = optionalAt! "name" text
  email = optionalAt! "email" text
  url = optionalAt! "url" text
  NpmHuman name email url

{{ Decode an NpmSignature from JSON. }}
npmSignatureDecoder : '{Decoder} NpmSignature
npmSignatureDecoder = do
  use object at!
  keyid = at! "keyid" text
  sig = at! "sig" text
  NpmSignature  keyid sig

{{ Encode an NpmSignature to JSON. }}
npmSignatureEncoder : NpmSignature -> Json
npmSignatureEncoder = cases
  (NpmSignature keyid sig) ->
    Json.object [
      ("keyid", Json.text keyid),
      ("sig", Json.text sig)
    ]

{{ Decode a list of NpmSignatures from JSON. }}
npmSignatureListDecoder : '{Decoder} [NpmSignature]
npmSignatureListDecoder = Decoder.array npmSignatureDecoder

{{ Encode a list of NpmSignatures to JSON. }}
npmSignatureListEncoder : [NpmSignature] -> Json
npmSignatureListEncoder signatures =
  Json.array (map npmSignatureEncoder signatures)

{{ Decode an NpmDist from JSON. }}
npmDistDecoder : '{Decoder} NpmDist
npmDistDecoder = do
  use object at!
  use object optionalAt!
  tarball = at! "tarball" text
  shasum = at! "shasum" text
  integrity = optionalAt! "integrity" text
  fileCount = optionalAt! "fileCount" Decoder.nat
  unpackedSize = optionalAt! "unpackedSize" Decoder.nat
  signatures = optionalAt! "signatures" (Decoder.array npmSignatureDecoder)
  NpmDist tarball shasum integrity fileCount unpackedSize signatures

{{ Encode an NpmDist to JSON. }}
npmDistEncoder : NpmDist -> Json
npmDistEncoder = cases
  (NpmDist tarball shasum integrity fileCount unpackedSize signatures) ->
    use object empty
    empty
      |> addText "tarball" tarball
      |> addText "shasum" shasum
      |> addOptionalText "integrity" integrity
      |> addOptionalNat "fileCount" fileCount
      |> addOptionalNat "unpackedSize" unpackedSize
      |> addOptionalList "signatures" (Optional.map (map npmSignatureEncoder) signatures)
