# npm Registry Types

Sets up the `npm-registry-types` project with types modelling the npm registry
wire protocol.

```ucm :hide
npm-registry-types/main> builtins.mergeio
npm-registry-types/main> lib.install @unison/base/releases/latest
npm-registry-types/main> lib.install @ceedubs/json/releases/1.4.2
```

## Shared Types

Types shared across both abbreviated and full metadata formats.

```unison
{{ A person associated with a package — author, maintainer, or contributor.
The npm registry does not validate these fields; they come straight from
package.json at publish time. At least one field should be present. }}
type NpmHuman = { name : Optional Text, email : Optional Text, url : Optional Text }

{{ Cryptographic signature attached to a package version.
The ''keyid'' identifies the signing key; ''sig'' is the PGP signature of
''<package>@<version>:<integrity>''. }}
type NpmSignature = { keyid : Text, sig : Text }

{{ Distribution metadata for a specific package version.
Generated by the registry at publish time. The ''tarball'' URL is the
canonical location to download the package archive; the client uses
''integrity'' (SRI hash) or ''shasum'' (SHA-1) to verify the download. }}
type NpmDist =
  { tarball : Text
  , shasum : Text
  , integrity : Optional Text
  , fileCount : Optional Nat
  , unpackedSize : Optional Nat
  , signatures : Optional [NpmSignature]
  }

{{ Structured funding information with an optional type label. }}
type NpmFundingInfo = { fundingType : Optional Text, fundingUrl : Text }

{{ Funding information for a package, as declared in package.json.
Can be a single URL string or a structured object with type + url. }}
type NpmFunding
  = NpmFunding.Url Text
  | NpmFunding.Object NpmFundingInfo

{{ Repository reference as declared in package.json. }}
type NpmRepository = { repoType : Optional Text, repoUrl : Text, directory : Optional Text }
```

## Abbreviated Metadata

Returned when the client sends `Accept: application/vnd.npm.install-v1+json`.
This is the format `npm install` actually requests. It's much smaller than the
full packument — for popular packages, full metadata can exceed 10MB
uncompressed, while abbreviated metadata is typically <100KB.

```unison
{{ Abbreviated version object — the minimal fields needed to install
a specific version. The ''name'', ''version'', and ''dist'' fields are always
present. All others are omitted when irrelevant for the version. }}
type NpmAbbreviatedVersion =
  { name : Text
  , version : Text
  , dist : NpmDist
  , dependencies : Map Text Text
  , devDependencies : Map Text Text
  , optionalDependencies : Map Text Text
  , peerDependencies : Map Text Text
  , peerDependenciesMeta : Map Text (Map Text Boolean)
  , bundleDependencies : Optional [Text]
  , bin : Optional (Map Text Text)
  , engines : Optional (Map Text Text)
  , deprecated : Optional Text
  , hasShrinkwrap : Optional Boolean
  , hasInstallScript : Optional Boolean
  , cpu : Optional [Text]
  , os : Optional [Text]
  , funding : Optional NpmFunding
  }

{{ Abbreviated packument — the top-level document returned for an
abbreviated metadata request. Contains dist-tags for resolving
"latest" / named tags, and a map of all versions. }}
type NpmAbbreviatedPackument =
  { name : Text
  , modified : Text
  , distTags : Map Text Text
  , versions : Map Text NpmAbbreviatedVersion
  }
```

## Full Metadata

Returned when the client omits the Accept header or sends
`Accept: application/json`. Includes everything in the abbreviated format plus
readme text, publish timestamps, user stars, maintainer lists, and the complete
package.json fields for each version.

```unison
{{ Full version object — contains all abbreviated fields plus the
hoisted package.json fields and publish metadata. }}
type NpmFullVersion =
  { name : Text
  , version : Text
  , dist : NpmDist
  , dependencies : Map Text Text
  , devDependencies : Map Text Text
  , optionalDependencies : Map Text Text
  , peerDependencies : Map Text Text
  , peerDependenciesMeta : Map Text (Map Text Boolean)
  , bundleDependencies : Optional [Text]
  , bin : Optional (Map Text Text)
  , engines : Optional (Map Text Text)
  , deprecated : Optional Text
  , hasShrinkwrap : Optional Boolean
  , hasInstallScript : Optional Boolean
  , cpu : Optional [Text]
  , os : Optional [Text]
  , funding : Optional NpmFunding
  , description : Optional Text
  , keywords : Optional [Text]
  , homepage : Optional Text
  , bugs : Optional Text
  , license : Optional Text
  , author : Optional NpmHuman
  , contributors : Optional [NpmHuman]
  , maintainers : Optional [NpmHuman]
  , repository : Optional NpmRepository
  , main : Optional Text
  , scripts : Optional (Map Text Text)
  , readme : Optional Text
  , readmeFilename : Optional Text
  , nodeVersion : Optional Text
  , npmVersion : Optional Text
  , npmUser : Optional NpmHuman
  }

{{ Full packument — the complete package metadata document.
The ''time'' map records when each version was published, plus
synthetic ''created'' and ''modified'' keys. The ''users'' map records
npm usernames of users who have starred the package. }}
type NpmFullPackument =
  { name : Text
  , distTags : Map Text Text
  , versions : Map Text NpmFullVersion
  , time : Map Text Text
  , users : Optional (Map Text Boolean)
  , author : Optional NpmHuman
  , bugs : Optional Text
  , contributors : Optional [NpmHuman]
  , description : Optional Text
  , homepage : Optional Text
  , keywords : Optional [Text]
  , license : Optional Text
  , maintainers : Optional [NpmHuman]
  , readme : Optional Text
  , readmeFilename : Optional Text
  , repository : Optional NpmRepository
  }
```

## Accept Header

Determines which format the registry returns for `GET /{package}`.

```unison
{{ The content negotiation modes for requesting package metadata.
{NpmAcceptFormat.Abbreviated} yields {type NpmAbbreviatedPackument};
{NpmAcceptFormat.Full} yields {type NpmFullPackument}. }}
type NpmAcceptFormat
  = NpmAcceptFormat.Abbreviated
  | NpmAcceptFormat.Full

{{ Render the Accept header value for a given format. }}
NpmAcceptFormat.toHeader : NpmAcceptFormat -> Text
NpmAcceptFormat.toHeader = cases
  NpmAcceptFormat.Abbreviated -> "application/vnd.npm.install-v1+json"
  NpmAcceptFormat.Full        -> "application/json"
```

## Package Name Encoding

Scoped packages like `@scope/name` must be URL-encoded as `@scope%2Fname` when
used in registry URLs.

```unison
{{ The scope and name components of a scoped npm package. }}
type NpmScopedName = { scope : Text, packageName : Text }

{{ Represents an npm package name, which may be scoped. }}
type NpmPackageName
  = NpmPackageName.Unscoped Text
  | NpmPackageName.Scoped NpmScopedName

{{ Encode a package name for use in a registry URL path.
Scoped packages have their ''/'' encoded as ''%2F''. }}
NpmPackageName.toPath : NpmPackageName -> Text
NpmPackageName.toPath = cases
  NpmPackageName.Unscoped n -> n
  NpmPackageName.Scoped s   -> "@" ++ NpmScopedName.scope s ++ "%2F" ++ NpmScopedName.packageName s

{{ Construct the tarball download path for a given package + version.
The conventional pattern is ''/{name}/-/{unscoped-name}-{version}.tgz'',
but a proxy should prefer the ''dist.tarball'' URL from the packument
rather than constructing this, since the registry is authoritative. }}
NpmPackageName.tarballPath : NpmPackageName -> Text -> Text
NpmPackageName.tarballPath pkg version =
  use Text ++
  base = match pkg with
    NpmPackageName.Unscoped n -> n
    NpmPackageName.Scoped s   -> "@" ++ NpmScopedName.scope s ++ "%2F" ++ NpmScopedName.packageName s
  shortName = match pkg with
    NpmPackageName.Unscoped n -> n
    NpmPackageName.Scoped s   -> NpmScopedName.packageName s
  base ++ "/-/" ++ shortName ++ "-" ++ version ++ ".tgz"
```
